\section{Basic Number Theory}

We begin with some elementary facts about natural numbers.

\begin{definition}[Double]
\label{def:double}
For a natural number $n$, we define its \emph{double} as $\mathrm{double}(n) = 2n$.
\end{definition}

\begin{theorem}[Double is even]
\label{thm:double-even}
For any natural number $n$, $\mathrm{double}(n)$ is even.
\end{theorem}

\begin{proof}
By definition, $\mathrm{double}(n) = 2n$, which is divisible by $2$.
\end{proof}

\begin{theorem}[Double is monotone]
\label{thm:double-mono}
For natural numbers $m$ and $n$ with $m \le n$, we have $\mathrm{double}(m) \le \mathrm{double}(n)$.
\end{theorem}

\begin{proof}
Since $m \le n$, multiplying both sides by $2$ gives $2m \le 2n$.
\end{proof}

\begin{theorem}[Double add]
\label{thm:double-add}
For natural numbers $m$ and $n$, $\mathrm{double}(m + n) = \mathrm{double}(m) + \mathrm{double}(n)$.
\end{theorem}

\begin{proof}
We compute $2(m+n) = 2m + 2n$.
\end{proof}
